\newtheorem{defi}{定義}
\newtheorem{them}{定理}
\newtheorem{prop}[them]{命題}
\newtheorem{lemm}[them]{補題}
\newtheorem{cor}[them]{系}
\newtheorem{exam}{例}

\begin{document}

  \begin{them}
    自然数の二乗の逆数の和は次の実数に収束する:
    \[
      1 + \frac{1}{2^2} + \frac{1}{3^2} + \frac{1}{4^2} + \cdots = \frac{\pi^2}{6}
    \]
  \end{them}

  \begin{them}
    自然数の総和は次の実数に収束する:
    \[
      1 + 2 + 3 + 4 + \cdots = -\frac{1}{12}
    \]
  \end{them}


\end{document}
